<script>

let cart = [];
let total = 0;

let products = [];
let api = "https://script.google.com/macros/s/AKfycbzb8SeJorO47B2pHE4GePkxcWDJBy7G_xl1UFmZWizEYqxF1IZX9hrWB26JDwlJSlSZ/exec";

fetch(api)
  .then(res => res.json())
  .then(data => {
    products = data;
    renderProducts();
  });

function renderProducts(){

  let html = "";

  products.forEach(p => {

    html += `
    <div class="product">
      <img src="${p.image}">
      <h4>${p.name}</h4>
      <p>৳${p.price}</p>
      <p>Stock: ${p.stock}</p>
      <button onclick="addToCart('${p.name}', ${p.price}, ${p.stock})">
        Add to Cart
      </button>
    </div>
    `;
  });

  document.getElementById("productList").innerHTML = html;
}

/* ADD TO CART */
function addToCart(name, price, stock){

  let kg = prompt("কত পরিমাণ নিতে চান?");

  if(!kg || isNaN(kg)){
    alert("সঠিক পরিমাণ লিখুন");
    return;
  }

  kg = parseFloat(kg);

  if(kg > stock){
    alert("স্টক নেই!");
    return;
  }

  let itemTotal = price * kg;

  cart.push({
    name,
    kg,
    total: itemTotal
  });

  total += itemTotal;

  document.getElementById("count").innerText = cart.length;
  renderCart();
}

/* SHOW CART */
function renderCart(){

  let html = "";

  cart.forEach((item, index) => {
    html += `
    <div class="item">
      <span>${item.name} (${item.kg}kg) - ৳${item.total}</span>
      <button class="deleteBtn" onclick="removeItem(${index})">X</button>
    </div>
    `;
  });

  document.getElementById("items").innerHTML = html;
  document.getElementById("total").innerText = total;
}

/* REMOVE ITEM */
function removeItem(index){
  total -= cart[index].total;
  cart.splice(index,1);

  document.getElementById("count").innerText = cart.length;
  renderCart();
}

/* OPEN / CLOSE CART */
function openCart(){
  document.getElementById("cartModal").style.display = "block";
}

function closeCart(){
  document.getElementById("cartModal").style.display = "none";
}

/* SEND TO GOOGLE SHEET */
function sendToServer(data){

  let url = "https://script.google.com/macros/s/AKfycbxwG6fBxSKjdxLQfeD6LU7SdJoCHQ2P4uABMS6e-b-5_6hYKpARS0l7sPUnPCBWf-LSzQ/exec";

  return fetch(url, {   // 👈 return দিতে হবে
    method: "POST",
    body: JSON.stringify(data)
  });
}

/* CONFIRM ORDER */
function confirmOrder(){
console.log("Confirm clicked");

  let name = document.getElementById("name").value;
  let phone = document.getElementById("phone").value;
  let branch = document.getElementById("branch").value;
  let payment = document.getElementById("payment").value;
  let wallet = document.getElementById("wallet").value;
  let trxid = document.getElementById("trxid").value;

  if(!name || !phone || !branch || !payment){
    alert("সব তথ্য পূরণ করুন");
    return;
  }

  let itemsText = "";

  cart.forEach(i => {
    itemsText += `${i.name} - ${i.kg}kg - ৳${i.total}\n`;
  });

  let orderText =
`🛒 NEW ORDER

👤 ${name}
📞 ${phone}
🏪 ${branch}

📦 Items:
${itemsText}

💰 Total: ৳${total}

💳 ${payment}
🔢 ${trxid}
`;

  alert(orderText);

  sendToServer({
  name,
  phone,
  branch,
  payment,
  wallet,
  trxid,
  total,
  cart
})
.then(() => {

  alert("🎉 Order Successfully Placed!");

 // reload products AFTER update
  fetch(api + "?t=" + new Date().getTime())
    .then(res => res.json())
    .then(data => {
      products = data;
      renderProducts();
    });

});

  // RESET
  cart = [];
  total = 0;

  document.getElementById("items").innerHTML = "";
  document.getElementById("total").innerText = 0;
  document.getElementById("count").innerText = 0;

  closeCart();

fetch(api + "?t=" + new Date().getTime())
  .then(res => res.json())
  .then(data => {
    products = data;
    renderProducts();
  });
}
